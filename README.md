### 安装
```bash
npm install ljs-tree-table
or
yarn add ljs-tree-table
```

### 使用
```html
<template>
    <ljs-tree-table :datas="datas" :columns="columns"/>
</template>
```
```javascript
import ljsTreeTable from 'ljs-tree-table'
import Vue from 'vue'

Vue.use(ljsTreeTable)
```
